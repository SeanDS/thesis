%% Glossary/acronym definitions

\newglossaryentry{UTC}{name=UTC, description={Coordinated Universal Time}}
\newglossaryentry{HV}{name=HV, description={High voltage}}
\newglossaryentry{CDS}{name=CDS, description={LIGO Control and Data System}}
\newglossaryentry{ADC}{name=ADC, description={Analogue-to-digital converter}}
\newglossaryentry{CMRR}{name=CMRR, description={Common-mode rejection ratio}}
\newglossaryentry{AC}{name=ac, description={Alternating current, though typically this refers to non-zero frequencies}}
\newglossaryentry{DC}{name=dc, description={Direct current, though typically this refers to zero frequency, i.e. \SI{0}{\hertz}}}
\newglossaryentry{MOSFET}{name=MOSFET, description={Metal-oxide-semiconductor field-effect transistor, an electronic switch typically employed in high power applications}}
\newglossaryentry{LED}{name=LED, description={Light-emitting diode}}
\newglossaryentry{ESD}{name=ESD, description={Electrostatic drive, a type of actuator}}
